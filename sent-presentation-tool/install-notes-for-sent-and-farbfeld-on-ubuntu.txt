# Assuming ~/src for code directory ...

# Install farbfeld and sent dependencies
sudo apt-get install git make gcc libx11-dev libxft-dev libpng-dev libjpeg-dev

# Fetch sent and farbfeld
git clone http://git.suckless.org/sent
git clone http://git.suckless.org/farbfeld

# Patch and compile farbfeld
patch < farbfeld-adjust-for-ubuntu.patch
make

# Compile sent
make

# Run with:
PATH=$PATH:~/src/farbfeld ./sent example.txt
